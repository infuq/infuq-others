1.安装bochs
2.配置bochsrc文件
3.制作镜像
[root@infuq /] bximage

4.制作操作系统
以下命令在Ubuntu16-32bit环境下执行
nasm -I ./include/ mbr.asm -f bin -o mbr.bin
nasm -I ./include/ loader.asm -f bin -o loader.bin
nasm ./lib/kernel/print.asm -f elf -o print.o
gcc -I ./lib/kernel/ -c -o main.o main.c
ld -Ttext 0xc0001500 -e main -o kernel.bin main.o print.o

dd if=mbr.bin of=infuq-os.img bs=512 count=1 conv=notrunc
dd if=loader.bin of=infuq-os.img bs=512 count=4 seek=2  conv=notrunc
dd if=kernel.bin of=infuq-os.img bs=512 count=200 seek=9  conv=notrunc

5.运行
[root@infuq /] bochs -f bochsrc
